```java
import java.util.Scanner; // 스캐너 적으면 나온다

public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in); // 스캐너가 가능하게 적음
        long a = sc.nextLong(); // long이 되는지 한번 해봄 보통은 int로 사용
        long b = sc.nextLong();

        System.out.println(a + b);
    }
} // 출력을 하면 수를 입력할 수 있음 수를 입력하면 a에 한번 b에 한번 들어가면서 +한 값이 나옴
```

```java
import java.util.Scanner;
```

```java
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int A = sc.nextInt();
        int B = sc.nextInt();
        int C = sc.nextInt();

        System.out.println((A+B)%C);
        System.out.println(((A%C) + (B%C))%C);
        System.out.println((A*B)%C);
        System.out.println(((A%C) * (B%C))%C); // 간단한 문제 // 조건이 4자리이기에 int 사용

    }
}
```

```java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int b = sc.nextInt();

        System.out.println((b%10) * a); // 10%로 나누면 5가 남는다 그걸 a에 곱한다 // 5 * 472
        System.out.println(((b%100)/10)*a); //100 %로 나누면 85가 남는다 그걸 10으로 나누면 8이 남고 그걸 a에 곱한다 // 8 *472 
        System.out.println((b/100) *a); //100으로 나누면 3.85 .85는 날라간다 그걸 a에 곱한다 3 * 472
        System.out.println(a*b);
// a= 472, b = 385인데 b의 385를 하나하나씩 3,8,5로 곱하기 위해 머리를 써야했다
// 385 / 10을 하면 38.5 1의 자리수 5는 나누어 지지 않는 나머지 숫자다
// 나머지 숫자를 구하는걸 %로 한다
    }
}
```
