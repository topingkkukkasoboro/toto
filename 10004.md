```java
    public static void main(String[] args) {

        int n = sc.nextInt(); // n을 스캔해서 입력받자 // 정수 갯수가 될것
        int[]arr = new int[n]; // arr의 배열은 n의 수만큼의 배열 선언
        int count = 0; // 같은 숫자 저장소다

        for (int i = 0; i < n; i++) { // i가 n보다 작지않을떄까지 1씩더하고
            arr[i] = sc.nextInt(); // arr배열의 i번쨰 마다 스캔입력
            }
        int v = sc.nextInt(); // v는 같은 숫자 스캔

        for (int i = 0; i <n ; i++) { // i가 arr배열의 길이가 될때까지 // arr의 길이는  n이다
            if(v==arr[i]){ //v랑 arr의 i번째 숫자 와 같으면
                count++; // count에 1씩 더해라
            }
        }
        System.out.println(count); //count 출력
        }
    }
```

```java
import java.util.Arrays;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        int n = sc.nextInt();
        int[] a = new int[n];
        int max = Integer.MIN_VALUE; // int형에서의 최소 값 // 마이너스 숫자가 나오는 경우를 대비해서 가장 작은 숫자


        for (int i = 0; i < n; i++) {
            a[i] = sc.nextInt();
        }

        for (int i = 0; i < n; i++) {
            max = Math.max(a[i], max);   // // int형에서의 최대값 // 최대값으로 숫자를 다 비교할수있게
        }
        int min = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            min = Math.min(a[i], min);
        }
        System.out.println(min + " " + max);
    }
}
```

```java
// 확실하게 전날 2시간동안 고생한 보람이 있었다. 최소값 최대값구하는 법이 각인된거 같다
//이번문제는 최대 값을 구하고 그게 몇번째인지 나타내야 했다.
// 숫자는 0 부터 세아리니 d에 1을 부여해서 보이는건 1부터 보이게 했다.
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        int[] a = new int[9]; // 자연수를 부여
        int d = 1;

        for (int i = 0; i < 9; i++) {
            a[i] = sc.nextInt(); //  9 만큼의 랜덤숫자
        }
        int max = Integer.MIN_VALUE;
        for (int i = 0; i < 9; i++) {
            max = Math.max(a[i], max); // 최대값 구하기 // max에 a[0]을 넣어서 비교
        }
        for (int i = 0; i < 9; i++) {
            if (a[i] != max){ // 일단 0번쨰 자리부터 비교한다 같지 않으면 
                d ++; // d는 1이 있는데 1을 넣는다 // 왜? 0부터세아리기에 처음 비교한거를 첫번쨰자리와 비교한것처럼 보이게 해야했다다
            }else break;
        }
        System.out.println(max);
        System.out.println(d);
```
