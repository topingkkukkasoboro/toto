```java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int score = sc.nextInt();

        if (90 <= score && score <= 100) {
            System.out.println("A");
        }
        else if (80 <= score && score <= 89) {
            System.out.println("B");
        }
        else if (70 <= score && score <= 79) {
            System.out.println("C");
        }
        else if (60 <= score && score <= 69) {
            System.out.println("D");
        }
        else if (0 <= score && score <= 68) {
            System.out.println("F");
        }
    }
    }
// 여기서는 score의 값을 어떻게 출력하냐로 작성한 것이다 ex) 시험점수로 등급 메기기 - 1등만 기억하는 더러운 세상
//숫자로 조건을 제시했고 범위를 정했다 (90 <= score && score <= 100)
// 자바는 귀찮게 설명을 많이 해줘야 한다. 해석하면 (90 < = score) 90보다 크거나 같으면 (&&) 그리고 (score <= 100)
//score가 100보다 작거나 같다면 // 조건이 2가지인 경우 &&를 통해 그리고라는 표현을 해줘야한다
```

```java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int year = sc.nextInt();

// 윤년을 구하는 문제이다 조건이 까다로웠다 // 년수가 4의 배수이면서 100의 배수가 아닐때 혹은 400의 배수일때 윤년이다 윤년인지 확인 가능한 코드를 만들자
// 윤년이면 1 아니면 0이 나와야한다
        if (((year%4)==0 && (year%100)!=0) || (year%400)==0) {// 말을 이해 못해서 그대로 작성했다 // 나머지를 구하는 방식으로 참인지 거짓인지 구했다
// y를 %했을때 나머지가 0과 같으며 100을%했을때 0과 다르거나 400을 %했을때 나머지가 0과 같아야 한다
            System.out.println(1);
        } else{
            System.out.println(0);
        }

    }
}
```

``` java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt();
        int b = sc.nextInt();
// 4분면을 구하는 문제이다 a와 b의 수가 주어지면 해당 수는 어느 면에 해당하는지 구해야 했다
// 말 그대로 작성하면 되는 문제 였다
// 수는 양수와 음수로만 주어지기에 0으로 비교를 하면된다
        if (a > 0 && b > 0 ){
            System.out.println(1);
// a가 0보다 크고 b가 0보다 크다면 1을 출력해라
        }
        if (a<0 && b> 0){
            System.out.println(2);
// a가 0보다 작고 b가 0보다 크다면 2를 출력해라
        }
        if (a < 0 && b < 0){
            System.out.println(3);
//a가 0보다 작고 b도 0보다 작다면 3을 출력해라
        }
        if (a > 0 && b < 0){
            System.out.println(4);
// a가 0보다 크고 b가 0보다 작으면 4를 출력해
        }
    }
}
```
