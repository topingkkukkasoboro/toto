import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();

        // 구구단 문제 // 수가 주어지면 1~9까지 곱한 값을 출력하자
        for (int a = 1; a <= 9 ; a++){ // a에 1을 대입, a가 9와 같아질때 까지, a에 1씩 더하자
            System.out.printf("%d * %d = %d%n", n, a, n * a);
        }
        // %d를 사용하니 풀렸다.... %d를 사용하면 해당 위치에 숫자가 들어간다...
        // %n을 하면 줄바꿈이 생긴다
        // printf로 적어야지 작동을 한다
    }
}

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int T = scanner.nextInt();
// 문제를 이해 못했다 예제를 보고 T에 값을 넣으면 랜덤하게 더해야 하나 싶었다
// 내가 잘못 봤다

        for (int c = 0; c < T; c++) { // c에 0으로 초기화 해라 (그리고) c보다 T가 클때까지 (뭘?) c에다 1씩 플러스해라
            int A = scanner.nextInt(); // A의 값을 스캔
            int B = scanner.nextInt(); //B의 값을 스캔

            System.out.println(A + B); // A +B의 값 출력
        }
    }
}

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);
        int a = sc.nextInt(); // a는 1보다 크고 10000보다 같거나 작다
        int b = 0;
        int c = 0;  /  for문 {}안에 선언을 하면 외부에서 안 뽑히더라 // 밖에서 선언을 하자

        for (; b <= a; b++){ // b가 a와 같아질때까지 (뭘?) b에 1씩 더해라
            c = b + c; // c에다가 b + c를 대입해라 (언제까지?) b가 a와 같아질떄까지
        }
        System.out.println(c); // c 뽑아라
        }


    }
